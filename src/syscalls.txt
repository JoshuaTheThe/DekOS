0x00-0x0F: # This segment is process related #
	EXIT                @ebx=code, noreturn
	GET_MY_PID          @returns pid
	GET_PARENTS_PID     @returns pid
	YIELD               @ebx=pid/none
	CREATE_PROCESS      @ebx=path, ecx=argc, edx=argv, returns pid
	KILL_PROCESS        @ebx=pid
	CHECK_PROCESS       @ebx=pid, returns 1 ? 0
	RESUME_PROCESS      @ebx=pid
	SUSPEND_PROCESS     @ebx=pid
	USERNAME            @ebx=buf,ecx=sizeof(buf)
	TICKS               @returns kernel tick count
	...
	# Rest Reserved
0x10-0x1F: # This segment is I/O related #
	OPEN                @ebx=path,ecx=sizeof(path), returns struct<*_FILE_>
	CLOSE               @ebx=struct<*_FILE_>
	READ                @ebx=struct<*_FILE_>, ecx=buf, edx=len
	WRITE               @ebx=struct<*_FILE_>, ecx=buf, edx=len
        GETCH               @returns from stdin
        PUTCH               @ebx=chr

0x20-0x2F: # MISC // IPC #
	SEND
	FETCH
	UNREAD
        MOUSEX              @returns mouseX
        MOUSEY              @returns mouseY
        MOUSEBTN            @returns mouse button mask

0x30-0x3F: # Memory Management #
	MALLOC              @returns heap allocated ptr
	FREE                @ebx = p

0x40-0x4F:	# Resource
	RREQ		    @ebx = rid, returns the handle of a resource
	BLIT		    @ebx = handle, ecx = buf, edx = sizeof(buf), edi = bytes_to_copy, esi = off
        RBLIT               @ebx = handle, ecx = dst, edx = sizeof(dst), edi = bytes_to_copy, esi = off
	RNEW		    @ebx = type, ecx = size, returns the handle
	RDEL		    @ebx = handle
	RGIVE		    @ebx = handle, ecx = pid
        RSIZE               @ebx = handle, returns sizeof data
        RONHEAP             @ebx = handle, returns whether it is on the heap or not
        RTYPE               @ebx = handle, returns the type of the resource
        ROWNER              @ebx = handle, returns the owner of the resource
        RPROTECT            @ebx = handle, ecx = mask
